ProjectCard Component 
import { ProjectCard } from './src/components/ProjectCard.js';

const card = new ProjectCard({
  title: 'My Project',
  description: 'Built with modern tools.',
  tags: ['React', 'Node.js'],
  links: { demo: '#', source: '#' },
  featured: false
});

document.getElementById('grid').appendChild(card.render());
card.reveal();

// Update dynamically
card.update({ title: 'Updated Title' });

// Listen for hover
card.on('hover', ({ hovered }) => console.log(hovered));

ContactForm Component 
import { ContactForm } from './src/forms/ContactForm.js';

const form = new ContactForm({
  onSubmit: async (data) => {
    await fetch('/api/contact', {
      method: 'POST',
      body: JSON.stringify(data)
    });
  },
  successMessage: 'Thanks! Message sent.',
  fields: [
    { name: 'name', type: 'text', label: 'Name',
      required: true, minLength: 2 },
    { name: 'email', type: 'email', label: 'Email',
      required: true, email: true },
    { name: 'message', type: 'textarea', label: 'Message',
      required: true, minLength: 10, rows: 4 }
  ]
});

container.appendChild(form.render());

// Events
form.on('submitted', ({ data }) => console.log(data));
form.on('validationFailed', ({ fields }) => console.log(fields));

// Programmatic
form.setData({ name: 'Jane', email: 'jane@test.com' });
form.reset();
ThemeManager & ThemeToggle Utility

Current theme: loading... 

import { ThemeManager } from './src/theme/ThemeManager.js';

const theme = new ThemeManager({
  defaultTheme: 'light',
  respectSystemPreference: true,
  transitionDuration: 200
});

theme.toggle();
theme.setTheme('dark');
theme.isDark(); // true

// Register a custom theme
theme.registerTheme('ocean', {
  '--fc-primary': '#0ea5e9',
  '--fc-background': '#0c1222',
  '--fc-surface': '#162032',
  '--fc-text': '#e0f2fe',
  '--fc-text-muted': '#7dd3fc',
  '--fc-border': '#1e3a5f',
});
theme.setTheme('ocean');

// Listen for changes
theme.on('themeChange', ({ theme }) => {
  console.log('Now using:', theme);
});
EventEmitter Core

Open browser console to see events fire. 

import { EventEmitter } from './src/core/EventEmitter.js';

const emitter = new EventEmitter();

emitter.on('data', (payload) => {
  console.log('Received:', payload);
});

emitter.once('init', () => {
  console.log('Fired once only');
});

emitter.emit('data', { id: 1, value: 'hello' });
emitter.emit('init');
emitter.emit('init'); // won't fire

emitter.removeAllListeners();

createPortfolio() Builder

One function call builds an entire portfolio. The live demo is built entirely with createPortfolio().

Open Full Demo 
import { createPortfolio } from 'foliocraft';

const portfolio = createPortfolio({
  target: '#app',
  theme: 'developer',

  hero: {
    headline: 'I build things for the web.',
    cta: { text: 'View Work', href: '#fc-projects' },
    socials: [{ platform: 'github', url: '#' }]
  },

  stats: [
    { value: 50, label: 'Projects', suffix: '+' },
  ],

  projects: [
    { title: 'My App', description: 'Something cool.',
      tags: ['React'], links: { demo: '#' } }
  ],

  skills: [
    { title: 'Frontend',
      skills: [{ name: 'JS', level: 95, color: '#f0db4f' }] }
  ],

  experience: [
    { title: 'Senior Dev', subtitle: 'Acme',
      date: '2023', dateEnd: 'Present' }
  ],

  contact: {
    onSubmit: async (data) => await fetch('/api', {
      method: 'POST', body: JSON.stringify(data)
    })
  }
});

// Change theme later
portfolio.setTheme('bold');
